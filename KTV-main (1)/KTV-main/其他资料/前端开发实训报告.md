**2.3 状态管理（Pinia）**

实现认证状态管理：

```typescript
// stores/auth.ts
import { defineStore } from 'pinia'
import { ref } from 'vue'
import type { User } from '@/types'

export const useAuthStore = defineStore('auth', () => {
  const token = ref<string | null>(localStorage.getItem('token'))
  const user = ref<User | null>(JSON.parse(localStorage.getItem('user') || 'null'))
  const isLoggedIn = ref(!!token.value)

  function setAuth(newToken: string, newUser: User) {
    token.value = newToken
    user.value = newUser
    isLoggedIn.value = true
    localStorage.setItem('token', newToken)
    localStorage.setItem('user', JSON.stringify(newUser))
  }

  function logout() {
    token.value = null
    user.value = null
    isLoggedIn.value = false
    localStorage.removeItem('token')
    localStorage.removeItem('user')
  }

  return { token, user, isLoggedIn, setAuth, logout }
})
```

**2.4 API 接口封装**

实现 Axios 实例和拦截器：

```typescript
// api/client.ts
import axios from 'axios'

export const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || 'https://localhost:5001',
  timeout: 10000,
  headers: { 'Content-Type': 'application/json' },
})

// 请求拦截器 - 自动附加 Token
apiClient.interceptors.request.use((config) => {
  const token = localStorage.getItem('token')
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// 响应拦截器 - 处理 401 错误
apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('token')
      localStorage.removeItem('user')
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)
```

**3.2 API 接口封装**

封装小程序的网络请求：

```javascript
// utils/api.js
const API_BASE = 'https://localhost:5001/api'
const MEDIA_BASE = 'https://localhost:5001'

function request(url, method = 'GET', data = {}) {
  return new Promise((resolve, reject) => {
    const token = wx.getStorageSync('token')
    wx.request({
      url: API_BASE + url,
      method: method,
      data: data,
      header: {
        'Content-Type': 'application/json',
        'Authorization': token ? `Bearer ${token}` : ''
      },
      success: (res) => {
        if (res.statusCode === 200) {
          resolve(res.data)
        } else if (res.statusCode === 401) {
          wx.removeStorageSync('token')
          wx.removeStorageSync('user')
          wx.redirectTo({ url: '/pages/index/index' })
          reject(new Error('登录已过期'))
        } else {
          reject(res)
        }
      },
      fail: reject
    })
  })
}

module.exports = {
  API_BASE,
  MEDIA_BASE,
  get: (url) => request(url),
  post: (url, data) => request(url, 'POST', data),
  put: (url, data) => request(url, 'PUT', data),
  del: (url) => request(url, 'DELETE'),
}
```

```javascript
// pages/home/home.js
const api = require('../../utils/api')

Page({
  data: {
    searchText: '',
    activeTagIndex: 0,
    tags: ['全部', '流行', '摇滚', '民谣', '电子', 'R&B', '嘻哈', '古典'],
    songList: [],
    loading: false,
  },

  onLoad() {

  },

  // 获取歌曲列表
  async fetchSongs() {
    this.setData({ loading: true })
    try {
      const res = await api.get('/songs')
      this.setData({ songList: res.data, loading: false })
    } catch (err) {
      console.error('获取歌曲失败:', err)
      this.setData({ loading: false })
    }
  },

  // 搜索歌曲
  onSearchInput(e) {
    this.setData({ searchText: e.detail.value })
    this.fetchSongs()
  },

  // 切换分类
  onTagTap(e) {
    const index = e.currentTarget.dataset.index
    this.setData({ activeTagIndex: index })
    this.fetchSongs()
  },

  // 点歌
  async onOrderSong(e) {
    const songId = e.currentTarget.dataset.id
    try {
      await api.post('/rooms/order-song', { songId })
      wx.showToast({ title: '点歌成功', icon: 'success' })
    } catch (err) {
      wx.showToast({ title: '请先加入房间', icon: 'none' })
    }
  },
```

**3.6 房间内页面开发**

实现房间内播放队列和聊天功能：

```javascript
// pages/rooms/rooms.js
const api = require('../../utils/api')
const app = getApp()

Page({
  data: {
    roomCode: '',
    roomInfo: {},
    queue: [],
    currentTrack: null,
    chatMessages: [],
    chatInput: '',
    isPlaying: false,
  },

  onLoad(options) {
    this.setData({ roomCode: options.code })
    this.joinRoom()
  },

  // 加入房间
  async joinRoom() {
    try {
      const res = await api.post('/rooms/join', { roomCode: this.data.roomCode })
      this.setData({ roomInfo: res.data })
      this.startPolling()
    } catch (err) {
      wx.showToast({ title: '加入房间失败', icon: 'none' })
    }
  },

  // 轮询播放状态
  startPolling() {
    this.pollTimer = setInterval(async () => {
      try {
        const res = await api.get(`/rooms/${this.data.roomInfo.id}/state`)
        this.setData({
          queue: res.data.queue,
          currentTrack: res.data.currentTrack,
          isPlaying: res.data.isPlaying
        })
      } catch (err) {
        console.error('轮询失败:', err)
      }
    }, 2000)
  },

  // 发送聊天消息
  async sendChat() {
    if (!this.data.chatInput.trim()) return
    try {
      await api.post('/chat/send', {
        roomId: this.data.roomInfo.id,
        message: this.data.chatInput
      })
      this.setData({ chatInput: '' })
    } catch (err) {
      wx.showToast({ title: '发送失败', icon: 'none' })
    }
  },

  onUnload() {
    if (this.pollTimer) clearInterval(this.pollTimer)
  },
})
```

## 五、遇到的问题及解决方案

### 问题 1：Token 过期后页面未自动跳转登录

**问题描述**：JWT Token 过期后，用户操作时页面未自动跳转到登录页。

**解决方案**：
- PC端：在 Axios 响应拦截器中添加 401 错误处理
- 小程序：在 API 请求封装中检测 401 状态码，自动清除 Token 并跳转

```javascript
// 小程序解决方案
if (res.statusCode === 401) {
  wx.removeStorageSync('token')
  wx.removeStorageSync('user')
  wx.redirectTo({ url: '/pages/index/index' })
}
```

### 问题 2：多人播放不同步

**问题描述**：房间内多人播放时，播放进度不同步。

**解决方案**：使用轮询机制每 2 秒获取服务端播放状态，客户端校正时间漂移：

```javascript
// 小程序轮询方案
this.pollTimer = setInterval(async () => {
  const res = await api.get(`/rooms/${roomId}/state`)
  this.setData({
    currentTrack: res.data.currentTrack,
    isPlaying: res.data.isPlaying
  })
}, 2000)
```

### 问题 3：小程序自定义 TabBar 样式问题

**问题描述**：自定义 TabBar 在不同机型上显示异常。

**解决方案**：使用 `getSystemInfo` 获取状态栏高度，动态计算 TabBar 位置：

```javascript
onLoad() {
  const systemInfo = wx.getSystemInfoSync()
  this.setData({
    statusBarHeight: systemInfo.statusBarHeight
  })
}
```
